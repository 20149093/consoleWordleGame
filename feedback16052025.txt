def give_feedback(guess, secret):
    """

    Parameters
    ----------
    guess

    [X] H   [X] E   [X] L   [✓] L   [!] O

    """
    feedback = ""
    for i in range(5):
        letter = guess[i].upper()
        if guess[i] == secret[i]:
            feedback += f"[✓] {letter}   "
        elif guess[i] in secret:

        else:
